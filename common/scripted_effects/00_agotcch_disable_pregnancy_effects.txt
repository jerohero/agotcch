agotcch_add_wife_modifier = {
	add_character_modifier = {
		modifier = agotcch_wife_modifier
		years = 1
	}
	
	if = {
		limit = {
			is_pregnant = yes
		}

		remove_trait = pregnant
		end_pregnancy = yes
	}
}
